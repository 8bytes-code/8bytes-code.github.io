    bool isValid(string s) {

        int n = 0, n2 = 0;
        int len = s.length();                               


        for(int i = 0; i < len; i++){
            if(s[i]+1 == s[i+1] || s[i]+2 == s[i+1])                          
                n++;

            if(i < len/2 && (s[i]+1 == s[len-i-1] || s[i]+2 == s[len-i-1]))      
                n2++;
            
        }


        return (n == len/2 || n2 == len/2);
    }
};
